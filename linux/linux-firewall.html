# linux 防火墙

# 查看防火墙的状态

systemctl status firewalld

# 打开防火墙

sudo systemctl start firewalld

再次查看防火墙的状态

# 关闭防火墙

sudo systemctl stop firewalld

# 在防火墙开启时开放具体端口

sudo systemctl start firewalld
sudo firewall-cmd --permanent --add-port=80/tcp
sudo firewall-cmd --reload
sudo firewall-cmncd --query-port=80/tcp
